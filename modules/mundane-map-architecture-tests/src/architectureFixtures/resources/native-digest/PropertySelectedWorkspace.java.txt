package io.github.mundanej.map.nativeimage;

import java.security.MessageDigest;
import java.security.NoSuchAlgorithmException;

final class NativeShapefileWorkspace {
    static byte[] digest(byte[] bytes) throws NoSuchAlgorithmException {
        return MessageDigest.getInstance(System.getProperty("map.digest.algorithm")).digest(bytes);
    }
}
